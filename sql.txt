create table login (
    userId int not null AUTO_INCREMENT,
    username varchar(50) not null,
    password varchar(50) not null,
    PRIMARY KEY(userId)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE utf8mb4_unicode_ci;

create table series (
    userId int not null,
    id int not null AUTO_INCREMENT,
